def summary_statistics(data):
    mean = data.mean()
    variance = data.var(ddof=1)
    std_dev = data.std(ddof=1)
    minimum = data.min()
    maximum = data.max()
    data_range = maximum - minimum
    return [mean, variance, std_dev, minimum, maximum, data_range]

# 计算统计量
